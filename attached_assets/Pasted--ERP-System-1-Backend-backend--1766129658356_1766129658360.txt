```python
SQLAlchemy ORM ашиглана

from sqlalchemy import create_engine
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker
from .config import settings

# SQLite холболт
engine = create_engine(
    settings.DATABASE_URL,
    connect_args={"check_same_thread": False}  # SQLite-д хэрэгтэй
)

# Session үүсгэгч
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

# Бүх model-ийн base class
Base = declarative_base()

def get_db():
    """
    Database session авах
    FastAPI Depends-д ашиглана
    """
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()
```

```python
# backend/app/utils/security.py
"""
🔐 Security - JWT, Password хэшлэх
"""
from datetime import datetime, timedelta
from typing import Optional
from jose import JWTError, jwt
from passlib.context import CryptContext
from fastapi import Depends, HTTPException, status
from fastapi.security import OAuth2PasswordBearer
from sqlalchemy.orm import Session
from ..config import settings
from ..database import get_db
from ..models.user import User

# Password hashing
pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")
oauth2_scheme = OAuth2PasswordBearer(tokenUrl="/api/auth/login")

def hash_password(password: str) -> str:
    """Password hash болгох"""
    return pwd_context.hash(password)

def verify_password(plain: str, hashed: str) -> bool:
    """Password шалгах"""
    return pwd_context.verify(plain, hashed)

def create_access_token(data: dict, expires_delta: Optional[timedelta] = None) -> str:
    """JWT token үүсгэх"""
    to_encode = data.copy()
    expire = datetime.utcnow() + (expires_delta or timedelta(minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES))
    to_encode.update({"exp": expire})
    return jwt.encode(to_encode, settings.SECRET_KEY, algorithm=settings.ALGORITHM)

def get_current_user(
    token: str = Depends(oauth2_scheme),
    db: Session = Depends(get_db)
) -> User:
    """Token-оос хэрэглэгч авах"""
    credentials_exception = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Token хүчингүй",
        headers={"WWW-Authenticate": "Bearer"},
    )
    
    try:
        payload = jwt.decode(token, settings.SECRET_KEY, algorithms=[settings.ALGORITHM])
        email: str = payload.get("sub")
        if email is None:
            raise credentials_exception
    except JWTError:
        raise credentials_exception
    
    user = db.query(User).filter(User.email == email).first()
    if user is None:
        raise credentials_exception
    
    return user
```

```python
# backend/app/routers/auth.py
"""
🔐 Authentication Router
"""
from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.orm import Session
from datetime import datetime, timedelta
from pydantic import BaseModel, EmailStr
import secrets

from ..database import get_db
from ..models.user import User, PasswordResetRequest
from ..models.employee import Employee
from ..schemas.user import UserLogin, UserCreate, Token
from ..utils.security import hash_password, verify_password, create_access_token, get_current_user

router = APIRouter(prefix="/api/auth", tags=["Auth"])

# === Schemas ===
class ForgotPasswordRequest(BaseModel):
    email: EmailStr

class VerifySecurityAnswer(BaseModel):
    email: EmailStr
    answer: str

class ResetPasswordRequest(BaseModel):
    token: str
    new_password: str

class SetSecurityQuestion(BaseModel):
    question: str
    answer: str

class ChangePasswordRequest(BaseModel):
    current_password: str
    new_password: str

# === Endpoints ===
@router.post("/login", response_model=Token)
def login(credentials: UserLogin, db: Session = Depends(get_db)):
    """Нэвтрэх"""
    user = db.query(User).filter(User.email == credentials.email).first()
    
    if not user or not verify_password(credentials.password, user.password):
        raise HTTPException(status_code=401, detail="Имэйл эсвэл нууц үг буруу")
    
    if not user.is_active:
        raise HTTPException(status_code=401, detail="Хэрэглэгч идэвхгүй")
    
    user.last_login = datetime.utcnow()
    db.commit()
    
    token = create_access_token(data={"sub": user.email})
    employee = db.query(Employee).filter(Employee.user_id == user.id).first()
    
    return {
        "access_token": token,
        "token_type": "bearer",
        "user": {
            "id": user.id,
            "email": user.email,
            "role": user.role,
            "employee_id": employee.id if employee else None,
            "employee_name": f"{employee.first_name} {employee.last_name}" if employee else None
        }
    }

@router.post("/register", response_model=Token)
def register(data: UserCreate, db: Session = Depends(get_db)):
    """Бүртгүүлэх"""
    if db.query(User).filter(User.email == data.email).first():
        raise HTTPException(status_code=400, detail="Энэ имэйл бүртгэлтэй")
    
    user = User(
        email=data.email,
        password=hash_password(data.password),
        role=data.role
    )
    db.add(user)
    db.commit()
    db.refresh(user)
    
    token = create_access_token(data={"sub": user.email})
    
    return {
        "access_token": token,
        "token_type": "bearer",
        "user": {"id": user.id, "email": user.email, "role": user.role}
    }

@router.get("/me")
def get_me(current_user: User = Depends(get_current_user), db: Session = Depends(get_db)):
    """Миний мэдээлэл"""
    employee = db.query(Employee).filter(Employee.user_id == current_user.id).first()
    
    return {
        "id": current_user.id,
        "email": current_user.email,
        "role": current_user.role,
        "employee_id": employee.id if employee else None,
        "employee_name": f"{employee.first_name} {employee.last_name}" if employee else None,
        "has_security_question": bool(current_user.security_question)
    }

# === Password Reset ===
@router.post("/forgot-password")
def forgot_password(data: ForgotPasswordRequest, db: Session = Depends(get_db)):
    """Нууц үг сэргээх эхлүүлэх"""
    user = db.query(User).filter(User.email == data.email).first()
    
    if not user:
        raise HTTPException(status_code=404, detail="Хэрэглэгч олдсонгүй")
    
    if not user.security_question:
        raise HTTPException(status_code=400, detail="Нууц асуулт тохируулаагүй")
    
    return {"email": user.email, "security_question": user.security_question}

@router.post("/verify-security-answer")
def verify_security_answer(data: VerifySecurityAnswer, db: Session = Depends(get_db)):
    """Нууц асуултын хариулт шалгах"""
    user = db.query(User).filter(User.email == data.email).first()
    
    if not user:
```

```python
    db.commit()
    
    return {
        "message": "Бүртгэгдлээ",
        "work_hours": record.work_hours,
        "overtime_hours": record.overtime_hours
    }
```

```python
@router.post("/payroll/calculate")
def calculate_payroll(
    data: PayrollCreate,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """Цалин тооцоолох"""
    if current_user.role not in ["Admin", "HR", "Finance"]:
        raise HTTPException(status_code=403, detail="Эрхгүй")
    
    emp = db.query(Employee).filter(Employee.id == data.employee_id).first()
    if not emp:
        raise HTTPException(status_code=404, detail="Ажилтан олдсонгүй")
    
    # Давхардсан эсэх
    existing = db.query(Payroll).filter(
        Payroll.employee_id == data.employee_id,
        Payroll.period_start == data.period_start,
        Payroll.period_end == data.period_end
    ).first()
    
    if existing:
        raise HTTPException(status_code=400, detail="Тооцоологдсон")
    
    # Илүү цаг
    overtime = db.query(Attendance).filter(
        Attendance.employee_id == data.employee_id,
        Attendance.date >= data.period_start,
        Attendance.date <= data.period_end
    ).all()
    
    total_overtime = sum(r.overtime_hours or 0 for r in overtime)
    hourly_rate = emp.hourly_rate or (emp.base_salary / 176)
    overtime_pay = total_overtime * hourly_rate * 1.5
    
    gross = emp.base_salary + overtime_pay
    
    # Суутгал
    tax = gross * 0.10
    social = gross * 0.115
    health = gross * 0.02
    net = gross - (tax + social + health)
    
    payroll = Payroll(
        employee_id=data.employee_id,
        period_start=data.period_start,
        period_end=data.period_end,
        base_salary=emp.base_salary,
        overtime_pay=round(overtime_pay, 2),
        gross_salary=round(gross, 2),
        tax=round(tax, 2),
        social_insurance=round(social, 2),
        health_insurance=round(health, 2),
        net_salary=round(net, 2),
        status="Pending"
    )
    
    db.add(payroll)
    db.commit()
    
    return {"id": payroll.id, "gross_salary": gross, "net_salary": net}
```

```python
async def upload_document(
    file: UploadFile = File(...),
    name: str = Form(...),
    category_id: Optional[int] = Form(None),
    description: Optional[str] = Form(""),
    requires_signature: bool = Form(False),
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    ext = get_ext(file.filename)
    if ext not in ALLOWED_EXTENSIONS:
        raise HTTPException(status_code=400, detail="Файлын төрөл зөвшөөрөгдөөгүй")
    
    unique_name = f"{uuid.uuid4().hex}.{ext}"
    file_path = os.path.join(UPLOAD_DIR, unique_name)
    
    with open(file_path, "wb") as f:
        shutil.copyfileobj(file.file, f)
    
    file_size = os.path.getsize(file_path) / 1024
    
    doc = Document(
        category_id=category_id,
        uploaded_by=current_user.id,
        name=name,
        original_filename=file.filename,
        file_path=file_path,
        file_type=ext,
        file_size=round(file_size, 2),
        mime_type=MIME_TYPES.get(ext, 'application/octet-stream'),
        description=description,
        requires_signature=requires_signature
    )
    
    db.add(doc)
    db.commit()
    db.refresh(doc)
    
    return {"id": doc.id, "name": doc.name, "file_size": doc.file_size}

@router.post("/{document_id}/sign-on-document")
async def sign_on_document(
    document_id: int,
    signature_data: str = Form(...),
    position_x: float = Form(70),
    position_y: float = Form(80),
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """Файл дээр гарын үсэг overlay хийх"""
    doc = db.query(Document).filter(Document.id == document_id).first()
    if not doc:
        raise HTTPException(status_code=404, detail="Олдсонгүй")
    
    if doc.file_type not in ['png', 'jpg', 'jpeg']:
        # Зураг биш файлд зөвхөн record хадгална
        sig = DocumentSignature(
            document_id=document_id,
            user_id=current_user.id,
            signature_image=signature_data,
            signature_type="drawn",
            position_x=position_x,
            position_y=position_y
        )
        db.add(sig)
        doc.is_signed = True
        db.commit()
        return {"message": "Гарын үсэг бүртгэгдлээ", "type": "record_only"}
    
    try:
        # Зураг файлд overlay хийх
        original = Image.open(doc.file_path)
        
        sig_data = signature_data.split(',')[1] if ',' in signature_data else signature_data
        sig_bytes = base64.b64decode(sig_data)
        sig_img = Image.open(io.BytesIO(sig_bytes))
        
        if sig_img.mode != 'RGBA':
            sig_img = sig_img.convert('RGBA')
        
        # Хэмжээ (20%)
        sig_width = int(original.width * 0.2)
        sig_height = int(sig_img.height * (sig_width / sig_img.width))
        sig_img = sig_img.resize((sig_width, sig_height), Image.Resampling.LANCZOS)
        
        # Байрлал
        x = int((position_x / 100) * original.width)
        y = int((position_y / 100) * original.height)
        
        if original.mode != 'RGBA':
            original = original.convert('RGBA')
        
        original.paste(sig_img, (x, y), sig_img)
        
        # Хадгалах
        signed_name = f"signed_{uuid.uuid4().hex}.{doc.file_type}"
        signed_path = os.path.join(UPLOAD_DIR, signed_name)
        
        if doc.file_type in ['jpg', 'jpeg']:
            original = original.convert('RGB')
        
        original.save(signed_path)
        
        doc.file_path = signed_path
        doc.is_signed = True
        doc.version += 1
        
        sig = DocumentSignature(
            document_id=document_id,
            user_id=current_user.id,
            signature_image=signature_data,
            signature_type="overlay",
            position_x=position_x,
            position_y=position_y
        )
        db.add(sig)
        db.commit()
        
        return {"message": "Файл дээр гарын үсэг зурагдлаа", "type": "overlay"}
        
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
```

```python
@router.get("/stats")

    """Статистик"""
    today = datetime.utcnow().date()
    today_start = datetime.combine(today, datetime.min.time())
    today_end = today_start + timedelta(days=1)
    
    total_emp = db.query(Employee).filter(Employee.is_active == True).count()
    
    present = db.query(Attendance).filter(
        Attendance.date >= today_start,
        Attendance.date < today_end,
        Attendance.status.in_(["Present", "Late"])
    ).count()
    
    pending_leaves = db.query(LeaveRequest).filter(LeaveRequest.status == "Pending").count()
    pending_payroll = db.query(Payroll).filter(Payroll.status == "Pending").count()
    
    return {
        "total_employees": total_emp,
        "present_today": present,
        "absent_today": max(0, total_emp - present),
        "attendance_rate": round((present / total_emp * 100) if total_emp > 0 else 0, 1),
        "pending_leaves": pending_leaves,
        "pending_payroll": pending_payroll
    }
```

```python
        admin = db.query(User).filter(User.email == "admin@company.com").first()
        if not admin:
            admin = User(
                email="admin@company.com",
                password=hash_password("admin123"),
                role="Admin",
                security_question="Таны анхны тэжээвэр амьтны нэр?",
                security_answer="test"
            )
            db.add(admin)
            db.commit()
            db.refresh(admin)
            
            admin_emp = Employee(
                company_id=company.id,
                user_id=admin.id,
                employee_code="EMP000001",
                first_name="System",
                last_name="Admin",
                position="Administrator",
                hire_date=datetime.utcnow()
            )
            db.add(admin_emp)
            db.commit()
            print("✅ Admin: admin@company.com / admin123")
        
        # Document Categories
        cats = [("CONTRACT", "Гэрээ"), ("ID", "Иргэний үнэмлэх"), ("OTHER", "Бусад")]
        for code, name in cats:
            if not db.query(DocumentCategory).filter(DocumentCategory.code == code).first():
                db.add(DocumentCategory(name=name, code=code))
        db.commit()
        print("✅ Document categories үүсгэлээ")
        
    finally:
        db.close()
```
